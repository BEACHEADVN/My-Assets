- name: Install Stuff
        run: |
          sudo add-apt-repository -y ppa:ubuntu-toolchain-r/test 
          sudo apt-get update
          sudo apt-get install -y aria2 jq rclone sshpass python-is-python3 wget python3 lz4 xz-utils device-tree-compiler zlib1g-dev gcc g++ libc6 libstdc++6 python3-pip dialog libgtk-3-dev aapt busybox zip erofs-utils unzip p7zip-full zipalign zstd bc android-sdk-libsparse-utils xmlstarlet
          sudo pip3 install ConfigObj
          sudo pip3 install telebot
          sudo pip3 install setuptools

          pip3 install requests beautifulsoup4

          echo "📦 Installing xmlstarlet..."

          if ! command -v xmlstarlet >/dev/null 2>&1; then
            sudo add-apt-repository universe -y || true
            sudo apt-get update

            if sudo apt-get install -y xmlstarlet; then
              echo "✅ Installed xmlstarlet via apt"
            else
              echo "⚠️ Apt failed, trying snap..."

              if sudo snap install xmlstarlet; then
                echo "✅ Installed xmlstarlet via snap"
                sudo ln -sf /snap/bin/xmlstarlet /usr/local/bin/xmlstarlet || true
              else
                echo "⚠️ Snap failed, building from source..."

                wget -qO xmlstarlet.tar.gz https://sourceforge.net/projects/xmlstar/files/xmlstarlet/1.6.1/xmlstarlet-1.6.1.tar.gz/download

                tar -xzf xmlstarlet.tar.gz
                cd xmlstarlet-1.6.1

                ./configure
                make -j$(nproc)

                sudo make install

                cd ..
                rm -rf xmlstarlet-1.6.1 xmlstarlet.tar.gz

                echo "✅ Installed xmlstarlet from source"
              fi
            fi
          fi

          xmlstarlet --version || true

          echo "⬇️ Đang tải Android Command Line Tools từ Google..."
          curl -sLo cmd-tools.zip "https://dl.google.com/android/repository/commandlinetools-linux-11076708_latest.zip"

          echo "📦 Đang cấu hình sdkmanager..."
          unzip -q cmd-tools.zip -d cmd-tools-dir
          mkdir -p android-sdk/cmdline-tools
          mv cmd-tools-dir/cmdline-tools android-sdk/cmdline-tools/latest

          echo "🤖 Đang tải Build-Tools 34.0.0 chính chủ từ Google Android SDK..."
          echo "y" | ./android-sdk/cmdline-tools/latest/bin/sdkmanager --sdk_root=$(pwd)/android-sdk "build-tools;34.0.0" "platform-tools" > /dev/null

          echo "$(pwd)/android-sdk/build-tools/34.0.0" >> $GITHUB_PATH
          echo "$(pwd)/android-sdk/platform-tools" >> $GITHUB_PATH

          sudo ldconfig
          export PATH="$(pwd)/android-sdk/build-tools/34.0.0:$PATH"

          echo "🔍 Kiểm tra zipalign..."
          which zipalign || true
          which aapt || true
          ldd android-sdk/build-tools/34.0.0/zipalign || true

          echo "✅ Zipalign hoạt động bình thường"
          sudo ln -sf "$(pwd)/android-sdk/build-tools/34.0.0/zipalign" /usr/local/bin/zipalign
          sudo ln -sf "$(pwd)/android-sdk/build-tools/34.0.0/aapt" /usr/local/bin/aapt

- name: Lấy Link ROM Tự Động
        run: |
          python3 - <<EOF
          import requests
          from bs4 import BeautifulSoup
          import os
          import re
          
          headers = {
              "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36"
          }
          
          target_device = "myron"
          target_region = "China"
          base_url = "https://hyperosroms.com"
          model_url = f"{base_url}/model/{target_device}"
          
          res = requests.get(model_url, headers=headers, timeout=15)
          if res.status_code != 200:
              print(f"❌ Không thể truy cập danh sách ROM thiết bị {target_device} vùng {target_region}")
              exit(1)
          
          soup = BeautifulSoup(res.text, 'html.parser')
          rows = soup.find_all('tr')
          sub_page_url = ""
          
          for row in rows:
              cells = row.find_all('td')
              if len(cells) >= 5:
                  region_text = cells[1].get_text(strip=True)
                  type_text = cells[2].get_text(strip=True).lower()
                  
                  if region_text == target_region and type_text == "recovery":
                      link_tag = cells[0].find('a') or cells[-1].find('a')
                      if link_tag and 'href' in link_tag.attrs:
                          sub_page_url = base_url + link_tag['href'] if link_tag['href'].startswith('/') else link_tag['href']
                          break
          
          if not sub_page_url:
              print(f"❌ Không tìm thấy sub-page ROM phù hợp cho {target_device} ({target_region})")
              exit(1)
          
          res_sub = requests.get(sub_page_url, headers=headers, timeout=15)
          soup_sub = BeautifulSoup(res_sub.text, 'html.parser')
          
          package_name = ""
          for li in soup_sub.find_all(['li', 'p']):
              text = li.get_text()
              if "Package Name:" in text:
                  package_name = text.split("Package Name:")[-1].strip()
                  break
          
          if not package_name:
              match = re.search(r'([\w-]+\.zip)', res_sub.text)
              if match:
                  package_name = match.group(1)
          
          if not package_name:
              print(f"❌ Không tìm thấy tên gói ZIP hợp lệ cho {target_device}")
              exit(1)
          
          version_code = sub_page_url.split('/')[-1]
          high_speed_url = f"https://bkt-sgp-miui-ota-update-alisgp.oss-ap-southeast-1.aliyuncs.com/{version_code}/{package_name}"
          
          with open(os.environ['GITHUB_ENV'], 'a') as f:
              f.write(f"AUTO_INPUT_URL={high_speed_url}\n")
              f.write(f"ROM_FILE={package_name}\n")
          EOF
