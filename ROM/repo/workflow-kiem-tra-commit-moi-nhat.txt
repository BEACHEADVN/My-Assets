name: Kiem tra commit moi nhat

on:
  workflow_dispatch: # Chỉ chạy thủ công

permissions:
  contents: write

jobs:
  update-history:
    runs-on: ubuntu-latest
    steps:
      # Bước 1: Checkout repo hiện tại để lấy thông tin commit mới nhất
      - name: Checkout Source Repo
        uses: actions/checkout@v4

      # Bước 2: Chỉ lấy mã commit ngắn (7 ký tự)
      - name: Get Latest Commit SHA
        id: info
        run: |
          SHA_SHORT=$(git rev-parse --short HEAD)
          echo "sha=$SHA_SHORT" >> $GITHUB_OUTPUT

      # Bước 3: Checkout repo My-Assets (Sử dụng GH_TOKEN)
      - name: Checkout My-Assets Repo
        uses: actions/checkout@v4
        with:
          repository: 'BEACHEADVN/My-Assets'
          token: ${{ secrets.GH_TOKEN }}
          path: 'my-assets'

      # Bước 4: Tạo thư mục và ghi đè mã commit vào file .txt
      - name: Update TxT File
        run: |
          # Đảm bảo tạo đầy đủ cây thư mục ROM/repo
          mkdir -p my-assets/ROM/repo
          
          # Ghi đè mã commit mới nhất vào file txt
          echo "${{ steps.info.outputs.sha }}" > my-assets/ROM/repo/latest-commit-stocktoolbuild.txt

      # Bước 5: Commit và Push thay đổi về repo My-Assets
      - name: Push to My-Assets
        run: |
          cd my-assets
          git config --global user.name "github-actions[bot]"
          git config --global user.email "github-actions[bot]@users.noreply.github.com"
          
          # Đưa file vào hàng chờ kiểm tra (stage) trước
          git add ROM/repo/latest-commit-stocktoolbuild.txt
          
          # Kiểm tra xem file stage lên có gì khác so với commit cũ không
          if git diff --cached --quiet; then
            echo "Không có thay đổi mới."
          else
            git commit -m "Update commit: ${{ steps.info.outputs.sha }}"
            git push origin main
          fi
