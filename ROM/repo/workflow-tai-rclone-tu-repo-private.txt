name: Inject Rclone Fetch Step

on:
  workflow_dispatch: # Kích hoạt thủ công bằng nút bấm trên GitHub Actions

permissions:
  contents: write # Cấp quyền ghi vào repository cho GITHUB_TOKEN mặc định

jobs:
  inject-step:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Repository
        uses: actions/checkout@main
        with:
          token: ${{ secrets.GH_TOKEN }}
          fetch-depth: 1 # Tối ưu tốc độ: Chỉ lấy commit mới nhất, bỏ qua toàn bộ lịch sử commit cũ

      - name: Download and Inject Step
        env:
          GH_TOKEN: ${{ secrets.GH_TOKEN }}
        run: |
          # 1. Tải nội dung đoạn step cần chèn từ repo My-Assets
          curl -s -H "Authorization: token $GH_TOKEN" \
            -H "Accept: application/vnd.github.v3.raw" \
            -L "https://api.github.com/repos/BEACHEADVN/My-Assets/contents/ROM/repo/tai-rclone-tu-repo-private.txt?ref=main" \
            -o step_to_inject.txt

          # 2. Chạy Python script để kiểm tra trùng lặp và xử lý thụt lề chuẩn xác
          python3 - << 'EOF'
          import os
          import sys

          build_yml_path = ".github/workflows/build.yml"
          inject_file_path = "step_to_inject.txt"

          if not os.path.exists(build_yml_path):
              print(f"Error: Không tìm thấy file {build_yml_path}")
              sys.exit(1)

          # Đọc nội dung file build.yml gốc để kiểm tra trước
          with open(build_yml_path, "r", encoding="utf-8") as f:
              build_content = f.read()

          # Kiểm tra xem step này đã được chèn vào trước đó chưa
          if "Fetch Secret rclone.conf to Root Directory" in build_content:
              print("Warning: Step 'Fetch Secret rclone.conf to Root Directory' đã tồn tại trong build.yml rồi. Bỏ qua không chèn lại nữa!")
              # Tạo một file tạm để báo cho step sau biết là không cần push
              with open("skip_push.txt", "w") as f:
                  f.write("skip")
              sys.exit(0)

          # Đọc nội dung đoạn step cần chèn
          with open(inject_file_path, "r", encoding="utf-8") as f:
              inject_content = f.read()

          # Đọc lại dưới dạng list dòng để xử lý vị trí chèn
          lines = build_content.splitlines(keepends=True)

          target_index = -1
          # Tìm vị trí dòng có chứa "- name: Set up Rclone"
          for i, line in enumerate(lines):
              if "- name: Set up Rclone" in line:
                  target_index = i
                  break

          if target_index == -1:
              print("Error: Không tìm thấy step '- name: Set up Rclone' trong build.yml")
              sys.exit(1)

          # Lấy chính xác số lượng khoảng trắng thụt lề của step đích
          line_target = lines[target_index]
          indent_size = len(line_target) - len(line_target.lstrip())
          base_indent = " " * indent_size
          
          formatted_lines = []
          for sub_line in inject_content.splitlines():
              stripped = sub_line.strip()
              if not stripped:
                  formatted_lines.append("")
                  continue
              
              # Căn chỉnh dựa trên cấu trúc chuẩn của step
              if stripped.startswith("- name:"):
                  formatted_lines.append(f"{base_indent}{stripped}")
              elif stripped.startswith("env:") or stripped.startswith("run:"):
                  formatted_lines.append(f"{base_indent}  {stripped}")
              else:
                  # Dành cho các dòng lệnh bên trong run: hoặc các key của env:
                  formatted_lines.append(f"{base_indent}    {stripped}")

          formatted_inject = "\n".join(formatted_lines) + "\n\n"

          # Chèn vào trước step "Set up Rclone"
          lines.insert(target_index, formatted_inject)

          # Ghi đè lại vào file build.yml
          with open(build_yml_path, "w", encoding="utf-8") as f:
              f.writelines(lines)

          print("Đã căn chỉnh thụt lề chuẩn và chèn thành công vào build.yml!")
          EOF

      - name: Commit and Push changes
        env:
          GH_TOKEN: ${{ secrets.GH_TOKEN }}
        run: |
          # Nếu file skip_push.txt tồn tại, kết thúc step luôn không push nữa
          if [ -f skip_push.txt ]; then
            echo "Không có thay đổi nào cần cập nhật lên repo."
            exit 0
          fi

          git config --local user.email "github-actions[bot]@users.noreply.github.com"
          git config --local user.name "github-actions[bot]"
          git add .github/workflows/build.yml
          git commit -m "Automated: Inject private rclone fetch step with precise indentation" || echo "Không có thay đổi nào cần commit"
          
          # Đổi URL sang định dạng sử dụng token để vượt qua lỗi 403
          git remote set-url origin "https://x-access-token:${GH_TOKEN}@github.com/${{ github.repository }}.git"
          git push origin main
