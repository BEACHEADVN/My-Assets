       cd $GITHUB_WORKSPACE/AnyKernel3
       zip -r "${{ steps.extract.outputs.file_name }}-AnyKernel3.zip" . -x "*.zip" ".git/*"
       echo "=== CHECK ZIP ==="
       ls -lah

    - name: Get Commit Details
      id: commit_info
      run: |
        # Lấy mã hash ngắn (7 ký tự)
        SHORT_SHA=$(git rev-parse --short HEAD)
        # Lấy nội dung commit message
        MSG=$(git log -1 --pretty=format:"%s")
        
        echo "short_sha=$SHORT_SHA" >> $GITHUB_OUTPUT
        echo "msg=$MSG" >> $GITHUB_OUTPUT

    - name: Send Notification to Telegram
      if: success()
      uses: xireiki/channel-post@v1
      with:
        bot_token: ${{ secrets.TELEGRAM_BOT_TOKEN }}

        context: |
          <b>Repository:</b> <a href="https://github.com/${{ github.repository }}">${{ github.repository }}</a>
        parse_mode: HTML
        method: sendMessage

      with:
        bot_token: ${{ secrets.TELEGRAM_BOT_TOKEN }}

        context: |
